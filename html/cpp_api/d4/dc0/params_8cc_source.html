491 void Parameters::set_op_weights()
492 {
499  //
500  // first, count up the instances of each type of terminal.
501  //
502  int b_count = 0;
503  int c_count = 0;
504  int f_count = 0;
505  int z_count = 0;
506  int total_terms = 0;
507 
508  for (const auto& term : terminals)
509  {
510  switch (term->otype)
511  {
512  case 'b':
513  ++b_count;
514  break;
515  case 'c':
516  ++c_count;
517  break;
518  case 'f':
519  ++f_count;
520  break;
521  case 'z':
522  ++z_count;
523  break;
524  }
525  ++total_terms;
526  }
527  //
528  // next, calculate the operator weights.
529  // an operators weight is defined as
530  // 1/total_args * sum ([arg_type]_count/total_terminals)
531  // summed over each arg the operator takes
532  //
533  op_weights.clear();
534  int i = 0;
535  for (const auto& op : functions)
536  {
537  op_weights.push_back(0.0);
538  int total_args = 0;
539  for (auto& kv : op->arity)
540  {
541  switch (kv.first) // kv.first is the arity type (character)
542  {
543  case 'b':
544  for (unsigned j = 0; j < kv.second; ++j)
545  op_weights.at(i) += float(b_count)/float(total_terms);
546  break;
547  case 'c':
548  for (unsigned j = 0; j < kv.second; ++j)
549  op_weights.at(i) += float(c_count)/float(total_terms);
550  break;
551  case 'f':
552  for (unsigned j = 0; j < kv.second; ++j)
553  op_weights.at(i) += float(f_count)/float(total_terms);
554  break;
555  case 'z':
556  for (unsigned j = 0; j < kv.second; ++j)
557  op_weights.at(i) += float(z_count)/float(total_terms);
558  break;
559  }
560  total_args += kv.second;
561  }
562  op_weights.at(i) /= float(total_args);
563  ++i;
564  }
565  // Now, we need to account for the output types of the operators that have non-zero
566  // weights, in addition to the terminals.
567  // So we now upweight the terminals according to the output types of the terminals that have
568  // non-zero weights.
569 
570  int total_ops_terms = total_terms;
571  b_count = 0;
572  c_count = 0;
573  f_count = 0;
574  z_count = 0;
575  for (unsigned i = 0; i < functions.size(); ++i)
576  {
577  if (op_weights.at(i) > 0)
578  {
579  switch (functions.at(i)->otype)
580  {
581  case 'b':
582  ++b_count;
583  break;
584  case 'c':
585  ++c_count;
586  break;
587  case 'f':
588  ++f_count;
589  break;
590  case 'z':
591  ++z_count;
592  break;
593  }
594  }
595  ++total_ops_terms;
596  }
597  /* cout << "b_count: " << b_count << "\n" */
598  /* << "f_count: " << f_count << "\n" */
599  /* << "c_count: " << c_count << "\n" */
600  /* << "z_count: " << z_count << "\n" */
601  /* << "total_ops_terms: " << total_ops_terms << "\n"; */
602 
603  i = 0; // op_weights counter
604  for (const auto& op : functions)
605  {
606  int total_args = 0;
607  for (auto& kv : op->arity)
608  {
609  switch (kv.first) // kv.first is the arity type (character)
610  {
611  case 'b':
612  for (unsigned j = 0; j < kv.second; ++j)
613  op_weights.at(i) += float(b_count)/float(total_ops_terms);
614  break;
615  case 'c':
616  for (unsigned j = 0; j < kv.second; ++j)
617  op_weights.at(i) += float(c_count)/float(total_ops_terms);
618  break;
619  case 'f':
620  for (unsigned j = 0; j < kv.second; ++j)
621  op_weights.at(i) += float(f_count)/float(total_ops_terms);
622  break;
623  case 'z':
624  for (unsigned j = 0; j < kv.second; ++j)
625  op_weights.at(i) += float(z_count)/float(total_ops_terms);
626  break;
627  }
628  total_args += kv.second;
629  }
630  op_weights.at(i) /= float(total_args);
631  ++i;
632  }
633 
634  /* string ow = "op_weights: "; */
635  /* for (unsigned i = 0; i< functions.size(); ++i) */
636  /* ow += "(" + functions.at(i)->name + ", " + std::to_string(op_weights.at(i)) + "), "; */
