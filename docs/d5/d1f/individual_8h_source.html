220  {
221  /* Check whether this individual dominates b.
222  *
223  * Input:
224  *
225  * b: another individual
226  *
227  * Output:
228  *
229  * 1: this individual dominates b; -1: b dominates this; 0: neither dominates
230  */
231 
232  int flag1 = 0, // to check if this has a smaller objective
233  flag2 = 0; // to check if b has a smaller objective
234 
235  for (int i=0; i<obj.size(); ++i) {
236  if (obj[i] < b.obj[i])
237  flag1 = 1;
238  else if (obj[i] > b.obj[i])
239  flag2 = 1;
240  }
241 
242  if (flag1==1 && flag2==0)
243  // there is at least one smaller objective for this and none for b
244  return 1;
245  else if (flag1==0 && flag2==1)
246  // there is at least one smaller objective for b and none for this
247  return -1;
248  else
249  // no smaller objective or both have one smaller
250  return 0;
