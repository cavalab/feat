154 bool CMulticlassLogisticRegression::train_machine(CFeatures* data)
155 {
156  if (data)
157  set_features((CDotFeatures*)data);
158 
159  REQUIRE(m_features, "%s::train_machine(): No features attached!\n");
160  REQUIRE(m_labels, "%s::train_machine(): No labels attached!\n");
161  REQUIRE(m_labels->get_label_type()==LT_MULTICLASS, "%s::train_machine(): "
162  "Attached labels are no multiclass labels\n");
163  REQUIRE(m_multiclass_strategy, "%s::train_machine(): No multiclass strategy"
164  " attached!\n");
165 
166  int32_t n_classes = ((CMulticlassLabels*)m_labels)->get_num_classes();
167  int32_t n_feats = m_features->get_dim_feature_space();
168 
169  slep_options options = slep_options::default_options();
170  if (m_machines->get_num_elements()!=0)
171  {
172  SGMatrix<float64_t> all_w_old(n_feats, n_classes);
173  SGVector<float64_t> all_c_old(n_classes);
174  for (int32_t i=0; i<n_classes; i++)
175  {
176  CLinearMachine* machine = (CLinearMachine*)m_machines->get_element(i);
177  SGVector<float64_t> w = machine->get_w();
178  for (int32_t j=0; j<n_feats; j++)
179  all_w_old(j,i) = w[j];
180  all_c_old[i] = machine->get_bias();
181  SG_UNREF(machine);
182  }
183  options.last_result = new slep_result_t(all_w_old,all_c_old);
184  m_machines->reset_array();
185  }
186  options.tolerance = m_epsilon;
187  options.max_iter = m_max_iter;
188  slep_result_t result = slep_mc_plain_lr(m_features,(CMulticlassLabels*)m_labels,m_z,options);
189 
190  SGMatrix<float64_t> all_w = result.w;
191  SGVector<float64_t> all_c = result.c;
192  for (int32_t i=0; i<n_classes; i++)
193  {
194  SGVector<float64_t> w(n_feats);
195  for (int32_t j=0; j<n_feats; j++)
196  w[j] = all_w(j,i);
197  float64_t c = all_c[i];
198  CLinearMachine* machine = new CLinearMachine();
199  machine->set_w(w);
200  machine->set_bias(c);
201  m_machines->push_back(machine);
202  }
203  return true;
204 }
