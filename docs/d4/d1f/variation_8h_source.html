74  {
84  bool pass; // pass check for children undergoing variation
85  while (pop.size() < 2*params.pop_size)
86  {
87  Individual child; // new individual
88 
89  if ( r() < cross_rate) // crossover
90  {
91  // get random mom and dad
92  int mom = r.random_choice(parents);
93  int dad = r.random_choice(parents);
94  // create child
95  params.msg("crossing " + pop.individuals[mom].get_eqn() + " with " +
96  pop.individuals[dad].get_eqn(), 2);
97  pass = cross(pop.individuals[mom],pop.individuals[dad],child,params);
98 
99  params.msg("crossing " + pop.individuals[mom].get_eqn() + " with " +
100  pop.individuals[dad].get_eqn() + " produced " + child.get_eqn() + ", pass: "
101  + std::to_string(pass),2);
102  }
103  else // mutation
104  {
105  // get random mom
106  int mom = r.random_choice(parents);
107  params.msg("mutating " + pop.individuals[mom].get_eqn() + "(" +
108  pop.individuals[mom].program_str() + ")", 2);
109  // create child
110  pass = mutate(pop.individuals[mom],child,params);
111 
112  params.msg("mutating " + pop.individuals[mom].get_eqn() + " produced " +
113  child.get_eqn() + ", pass: " + std::to_string(pass),2);
114  }
115 
116 
117 
118  if (pass) // congrats! you produced a viable child.
119  {
120  // give child an open location in F
121  //child.loc = pop.get_open_loc();
122  //push child into pop
123  pop.add(child);

128  bool Variation::mutate(Individual& mom, Individual& child, const Parameters& params)
129  {
140  // make child a copy of mom
141  child.program = mom.program;
142 
143  float rf = r();
144  if (rf < 1.0/3.0 && child.get_dim() > 1){
145  delete_mutate(child,params);
146  assert(is_valid_program(child.program, params.num_features));
147  }
148  else if (rf < 2.0/3.0 && child.size() < params.max_size)
149  {
150  insert_mutate(child,params);
151  assert(is_valid_program(child.program, params.num_features));
152  }
153  else
154  {
155  point_mutate(child,params);
156  assert(is_valid_program(child.program, params.num_features));
157  }
158 
159  // check child depth and dimensionality
160  return child.size() <= params.max_size && child.get_dim() <= params.max_dim;

163  void Variation::point_mutate(Individual& child, const Parameters& params)
164  {
170  params.msg("\tpoint mutation",2);
171  float n = child.size();
172 
173  // loop thru child's program
174  for (auto& p : child.program)
175  {
176  if (r() < 1/n) // mutate p. TODO: change '1' to node weighted probability
177  {
178  params.msg("\t\tmutating node " + p->name, 2);
179  vector<std::shared_ptr<Node>> replacements; // potential replacements for p
180 
181  if (p->total_arity() > 0) // then it is an instruction
182  {
183  // find instructions with matching in/out types and arities
184  for (const auto& f: params.functions)
185  {
186  if (f->otype == p->otype && f->arity['f']==p->arity['f'] &&
187  f->arity['b']==p->arity['b'])
188  replacements.push_back(f);
189  }
190  }
191  else // otherwise it is a terminal
192  {
193  // TODO: add terminal weights here
194  // find terminals with matching output types
195 
196  for (const auto& t : params.terminals)
197  {
198  if (t->otype == p->otype)
199  replacements.push_back(t);
200  }
201  }
202  // replace p with a random one
203  p = r.random_choice(replacements);
204  }

325  bool Variation::cross(Individual& mom, Individual& dad, Individual& child,
326  const Parameters& params)
327  {
339  bool subtree = r() <0.5; // half the time, do subtree xo.
340  // half the time, swap dimensions.
341  vector<size_t> mlocs, dlocs; // mom and dad locations for consideration
342  size_t i1, j1, i2, j2; // i1-j1: mom portion, i2-j2: dad portion
343 
344  if (subtree)
345  {
346  params.msg("\tsubtree xo",2);
347  // limit xo choices to matching output types in the programs.
348  vector<char> d_otypes;
349  for (const auto& p : dad.program)
350  if(!in(d_otypes,p->otype))
351  d_otypes.push_back(p->otype);
352 
353  // get valid subtree locations
354  for (size_t i =0; i<mom.size(); ++i)
355  if (in(d_otypes,mom[i]->otype))
356  mlocs.push_back(i);
357  if (mlocs.size()==0) // mom and dad have no overlapping types, can't cross
358  {
359  std::cout << "\tno overlapping types between " + mom.program_str() + ","
360  + dad.program_str() + "\n";
361  return 0;
362  }
363  j1 = r.random_choice(mlocs);
364 
365  // get locations in dad's program that match the subtree type picked from mom
366  for (size_t i =0; i<dad.size(); ++i)
367  if (dad[i]->otype == mom[j1]->otype) dlocs.push_back(i);
368  }
369  else // half the time, pick a root node
370  {
371  params.msg("\troot xo",2);
372  mlocs = mom.roots();
373  dlocs = dad.roots();
374  params.msg("\t\trandom choice mlocs (size "+std::to_string(mlocs.size())+")",2);
375  j1 = r.random_choice(mlocs);
376  }
377  // get subtree
378  i1 = mom.subtree(j1);
379 
380  // get dad subtree
381  j2 = r.random_choice(dlocs);
382  i2 = dad.subtree(j2);
383 
384  // make child program by splicing mom and dad
385  child.program = splice_programs(mom.program, i1, j1, dad.program, i2, j2);
386 
387  if (params.verbosity >= 2)
388  print_cross(mom,i1,j1,dad,i2,j2,child);
389 
390  assert(is_valid_program(child.program,params.num_features));
391  // check child depth and dimensionality
392  return child.size() <= params.max_size && child.get_dim() <= params.max_dim;
393  }
