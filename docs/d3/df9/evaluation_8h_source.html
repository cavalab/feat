146  double Evaluation::bal_accuracy(const VectorXd& y, const VectorXd& yhat, vector<int> c,
147  bool reverse)
148  {
149  if (c.empty()) // determine unique class values
150  {
151  vector<double> uc = unique(y);
152  for (const auto& i : uc)
153  c.push_back(int(i));
154  }
155 
156  // sensitivity (TP) and specificity (TN)
157  vector<double> TP(c.size(),0.0), TN(c.size(), 0.0), P(c.size(),0.0), N(c.size(),0.0);
158  ArrayXd class_accuracies(c.size());
159 
160  // get class counts
161 
162  for (unsigned i=0; i< c.size(); ++i)
163  {
164  P.at(i) = (y.array().cast<int>() == c[i]).count(); // total positives for this class
165  N.at(i) = (y.array().cast<int>() != c[i]).count(); // total negatives for this class
166  }
167 
168 
169  for (unsigned i = 0; i < y.rows(); ++i)
170  {
171  if (yhat(i) == y(i)) // true positive
172  ++TP.at(y(i) == -1 ? 0 : y(i)); // if-then ? accounts for -1 class encoding
173 
174  for (unsigned j = 0; j < c.size(); ++j)
175  if ( y(i) !=c.at(j) && yhat(i) != c.at(j) ) // true negative
176  ++TN.at(j);
177 
178  }
179 
180  // class-wise accuracy = 1/2 ( true positive rate + true negative rate)
181  for (unsigned i=0; i< c.size(); ++i){
182  class_accuracies(i) = (TP[i]/P[i] + TN[i]/N[i])/2;
183  //std::cout << "TP(" << i << "): " << TP[i] << ", P[" << i << "]: " << P[i] << "\n";
184  //std::cout << "TN(" << i << "): " << TN[i] << ", N[" << i << "]: " << N[i] << "\n";
185  //std::cout << "class accuracy(" << i << "): " << class_accuracies(i) << "\n";
186  }
187  if (reverse)

189  else
