33  vector<size_t> Lexicase::select(const MatrixXd& F, const Parameters& params, Rnd& r)
34  {
35  /* Input:
36  *
37  * F: n_samples X popsize matrix of model outputs.
38  * params: parameters.
39  *
40  * Output:
41  *
42  * selected: vector of indices corresponding to columns of F that are selected.
43  */
44 
45  unsigned int N = F.rows(); //< number of samples
46  unsigned int P = F.cols(); //< number of individuals
47 
48  // define epsilon
49  ArrayXd epsilon = ArrayXd::Zero(N);
50 
51  // if output is continuous, use epsilon lexicase
52  if (!params.classification)
53  {
54  // for columns of F, calculate epsilon
55  for (int i = 0; i<epsilon.size(); ++i)
56  epsilon(i) = mad(F.row(i));
57  }
58 
59  // individual locations
60  vector<size_t> starting_pool(P);
61  iota(starting_pool.begin(), starting_pool.end(), 0);
62 
63  // cases
64  vector<size_t> cases(N);
65  iota(cases.begin(),cases.end(),0);
66 
67  vector<size_t> selected; // selected individuals
68 
69  for (unsigned int i = 0; i<P; ++i) // selection loop
70  {
71  vector<size_t> pool = starting_pool; // initial pool
72  vector<size_t> winner; // winners
73  r.shuffle(cases.begin(),cases.end()); // shuffle cases
74 
75  bool pass = true; // checks pool size and number of cases
76  unsigned int h = 0; // case count
77 
78  while(pass){ // main loop
79 
80  winner.resize(0); // winners
81  double minfit = std::numeric_limits<double>::max(); // minimum error on case
82 
83  for (size_t j = 0; j<pool.size(); ++j)
84  {
85  if (F(cases[h],pool[j]) < minfit+epsilon[cases[h]])
86  {
87  minfit=F(cases[h],pool[j]); // set the new minimum
88  winner.resize(0); // start the winners pool
89  winner.push_back(pool[j]);
90  }
91  else if (F(cases[h],pool[j]) == minfit+epsilon[cases[h]])
92  winner.push_back(pool[j]);
93  }
94 
95  ++h; // next case
96  pass = (winner.size()>1 && h<cases.size()); // only keep going if needed
97  pool = winner; // reduce pool to remaining individuals
98 
99  }
100  assert(winner.size()>0);
101  //if more than one winner, pick randomly
102  selected.push_back(r.random_choice(winner));
103  }
104 
105  return selected;
106  }
107 }
