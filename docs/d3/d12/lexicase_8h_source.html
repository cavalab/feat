31  vector<size_t> Lexicase::select(Population& pop, const MatrixXd& F, const Parameters& params)
32  {
33  /* Selection according to lexicase selection for classification and epsilon-lexicase
34  * selection for regression.
35  *
36  * Input:
37  *
38  * F: n_samples X popsize matrix of model outputs.
39  * params: parameters.
40  *
41  * Output:
42  *
43  * selected: vector of indices corresponding to columns of F that are selected.
44  * In selection mode, parents are selected among the first half of columns of F since
45  * it is assumed that we are selecting for offspring to fill the remaining columns.
46  */
47 
48  if (survival) // if survival is on, run lexicase survival.
49  return survive(pop,F,params);
50 
51  unsigned int N = F.rows(); //< number of samples
52  unsigned int P = F.cols()/2; //< number of individuals
53 
54  // define epsilon
55  ArrayXd epsilon = ArrayXd::Zero(N);
56 
57  // if output is continuous, use epsilon lexicase
58  if (!params.classification)
59  {
60  // for columns of F, calculate epsilon
61  for (int i = 0; i<epsilon.size(); ++i)
62  epsilon(i) = mad(F.row(i));
63  }
64 
65  // individual locations in F
66  vector<size_t> starting_pool;
67  for (const auto& p : pop.individuals) starting_pool.push_back(p.loc);
68  assert(starting_pool.size() == P);
69 
70  vector<size_t> F_locs(P,0); // selected individuals
71 
72  #pragma omp parallel for
73  for (unsigned int i = 0; i<P; ++i) // selection loop
74  {
75 
76  vector<size_t> cases(N); // cases
77  iota(cases.begin(),cases.end(),0);
78  vector<size_t> pool = starting_pool; // initial pool
79  vector<size_t> winner; // winners
80  r.shuffle(cases.begin(),cases.end()); // shuffle cases
81 
82  bool pass = true; // checks pool size and number of cases
83  unsigned int h = 0; // case count
84 
85  while(pass){ // main loop
86 
87  winner.resize(0); // winners
88  double minfit = std::numeric_limits<double>::max(); // minimum error on case
89 
90  for (size_t j = 0; j<pool.size(); ++j)
91  {
92  if (F(cases[h],pool[j]) < minfit+epsilon[cases[h]])
93  {
94  minfit=F(cases[h],pool[j]); // set the new minimum
95  winner.resize(0); // start the winners pool
96  winner.push_back(pool[j]);
97  }
98  else if (F(cases[h],pool[j]) == minfit+epsilon[cases[h]])
99  winner.push_back(pool[j]);
100  }
101 
102  ++h; // next case
103  pass = (winner.size()>1 && h<cases.size()); // only keep going if needed
104  pool = winner; // reduce pool to remaining individuals
105 
106  }
107 
108  assert(winner.size()>0);
109  //if more than one winner, pick randomly
110  F_locs[i] = r.random_choice(winner);
111  }
112 
113  // convert F_locs to pop.individuals indices
114  vector<size_t> selected;
115  bool match = false;
116  for (const auto& f: F_locs)
117  {
118  for (unsigned i=0; i < pop.size(); ++i)
119  {
120  if (pop.individuals[i].loc == f)
121  {
122  selected.push_back(i);
123  match = true;
124  }
125  }
126  if (!match)
127  std::cout << "no loc matching " << f << " in pop\n";
128  match = false;
129  }
130  if (selected.size() != F_locs.size()){
131  std::cout << "pop.locs: ";
132  for (auto i: pop.individuals) std::cout << i.loc << " "; std::cout << "\n";
133  std::cout << "selected: " ;
134  for (auto s: selected) std::cout << s << " "; std::cout << "\n";
135  std::cout<< "F_locs: ";
136  for (auto f: F_locs) std::cout << f << " "; std::cout << "\n";
137  }
138  assert(selected.size() == F_locs.size());
139  return selected;
140  }
