87  bool is_valid_program(vector<std::shared_ptr<Node>>& program, unsigned num_features)
88  {
90  vector<ArrayXd> stack_f;
91  vector<ArrayXb> stack_b;
92  MatrixXd X = MatrixXd::Zero(num_features,2);
93  VectorXd y = VectorXd::Zero(2);
94  unsigned i = 0;
95  for (const auto& n : program){
96  if ( stack_f.size() >= n->arity['f'] && stack_b.size() >= n->arity['b'])
97  n->evaluate(X, y, stack_f, stack_b);
98  else{
99  std::cout << "Error: ";
100  for (const auto& p: program) std::cout << p->name << " ";
101  std::cout << "is not a valid program because ";
102  std::cout << n->name << " at pos " << i << "is not satisfied\n";
103  return false;
104  }
105  ++i;
106  }
107  return true;
108  }

110  void make_tree(vector<std::shared_ptr<Node>>& program,
111  const vector<std::shared_ptr<Node>>& functions,
112  const vector<std::shared_ptr<Node>>& terminals, int max_d,
113  const vector<double>& term_weights, char otype)
114  {
115 
119  if (max_d == 0 || r.rnd_flt() < terminals.size()/(terminals.size()+functions.size()))
120  {
121  // append terminal
122  vector<size_t> ti, tw; // indices of valid terminals
123  for (size_t i = 0; i<terminals.size(); ++i)
124  {
125  if (terminals[i]->otype == otype) // grab terminals matching output type
126  {
127  ti.push_back(i);
128  tw.push_back(term_weights[i]);
129  }
130  }
131  auto t = terminals[r.random_choice(ti,tw)];
132  //std::cout << t->name << " ";
133  program.push_back(t);
134  }
135  else
136  {
137  // let fi be indices of functions whose output type matches otype and, if max_d==1,
138  // with no boolean inputs (assuming all input data is floating point)
139  vector<size_t> fi;
140  for (size_t i = 0; i<functions.size(); ++i)
141  if (functions[i]->otype==otype && (max_d>1 || functions[i]->arity['b']==0))
142  fi.push_back(i);
143  //std::cout << "fi size: " << fi.size() << "\n";
144  if (fi.size()==0){
145  std::cout << "---\n";
146  std::cout << "f1.size()=0. current program: ";
147  for (auto p : program) std::cout << p->name << " ";
148  std::cout << "\n";
149  std::cout << "otype: " << otype << "\n";
150  std::cout << "max_d: " << max_d << "\n";
151  std::cout << "functions: ";
152  for (auto f: functions) std::cout << f->name << " ";
153  std::cout << "\n";
154  std::cout << "---\n";
155  }
156  assert(fi.size() > 0 && "The operator set specified results in incomplete programs.");
157 
158  // append a random choice from fs
159  auto t = functions[r.random_choice(fi)];
160  //std::cout << t->name << " ";
161  program.push_back(t);
162 
163  std::shared_ptr<Node> chosen = program.back();
164  // recurse to fulfill the arity of the chosen function
165  for (size_t i = 0; i < chosen->arity['f']; ++i)
166  make_tree(program, functions, terminals, max_d-1, term_weights,'f');
167  for (size_t i = 0; i < chosen->arity['b']; ++i)
168  make_tree(program, functions, terminals, max_d-1, term_weights, 'b');
169 
170  }
