89  if (max_d == 0 || r.rnd_flt() < terminals.size()/(terminals.size()+functions.size()))

91  // append terminal

106  // type of float if max_d > 1 (assuming all input data is continouous)
107  vector<size_t> fi;
108  for (size_t i = 0; i<functions.size(); ++i)
109  {
110  if (functions[i]->otype == otype && (max_d>1 || functions[i]->arity['b']==0))
111  fi.push_back(i);
112  }
113  // append a random choice from fs
114  program.push_back(functions[r.random_choice(fi)]);
115 
116  std::shared_ptr<Node> chosen = program.back();
117  // recurse to fulfill the arity of the chosen function
118  for (size_t i = 0; i < chosen->arity['f']; ++i)
119  make_program(program, functions, terminals, max_d-1, 'f', term_weights);
120  for (size_t i = 0; i < chosen->arity['b']; ++i)
