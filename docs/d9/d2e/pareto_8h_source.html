60  bool check_pareto(Population& pop, vector<size_t> survivors)
61  {
63 
64  // get ranks of the population

66  unsigned front_num=0, s_front_num=0;
67  for (const auto& p: pop.individuals)
68  {

82  vector<size_t> Pareto::survive(Population& pop, const MatrixXd& F, const Parameters& params)
83  {
84  /* Selection using the survival scheme of NSGA-II.
85  *
86  * Input:
87  *
88  * pop: population of programs.
89  * params: parameters.
90  * r: random number generator
91  *
92  * Output:
93  *
94  * selected: vector of indices corresponding to columns of F that are selected.
95  * modifies individual ranks, objectives and dominations.
96  */
97 
98  // set objectives
99  #pragma omp parallel for
100  for (unsigned int i=0; i<pop.size(); ++i)
101  pop.individuals[i].set_obj(params.objectives);
102 
103  // fast non-dominated sort
104  fast_nds(pop);
105 
106  // Push back selected individuals until full
107  vector<size_t> selected;
108  int i = 0;
109  while ( selected.size() + front[i].size() < params.pop_size)
110  {
111  std::vector<int>& Fi = front[i]; // indices in front i
112  //crowding_distance(i); // calculate crowding in Fi
113 
114  for (int j = 0; j < Fi.size(); ++j) // Pt+1 = Pt+1 U Fi
115  selected.push_back(Fi[j]);
116 
117  ++i;
118  }
119 
120  crowding_distance(pop,i); // calculate crowding in final front to include
121  std::sort(front[i].begin(),front[i].end(),sort_n(pop));
122 
123  const int extra = params.pop_size - selected.size();
124  for (int j = 0; j < extra; ++j) // Pt+1 = Pt+1 U Fi[1:N-|Pt+1|]
125  selected.push_back(front[i][j]);
126 
127  assert(check_pareto(pop,selected));
128 
129  return selected;
130  }

132  void Pareto::fast_nds(Population& pop)
133  {
134  front.resize(1);
135  front[0].clear();
136  //std::vector< std::vector<int> > F(1);
137  #pragma omp parallel for
138  for (int i = 0; i < pop.size(); ++i) {
139 
140  std::vector<unsigned int> dom;
141  int dcount = 0;
142 
143  Individual& p = pop.individuals[i];
144  // p.dcounter = 0;
145  // p.dominated.clear();
146 
147  for (int j = 0; j < pop.size(); ++j) {
148 
149  Individual& q = pop.individuals[j];
150 
151  int compare = p.check_dominance(q);
152  if (compare == 1) { // p dominates q
153  //p.dominated.push_back(j);
154  dom.push_back(j);
155  } else if (compare == -1) { // q dominates p
156  //p.dcounter += 1;
157  dcount += 1;
158  }
159  }
160 
161  #pragma omp critical
162  {
163  p.dcounter = dcount;
164  p.dominated.clear();
165  p.dominated = dom;
166 
167 
168  if (p.dcounter == 0) {
169  p.set_rank(1);
170  front[0].push_back(i);
171  }
172  }
173 
174  }
175 
176  // using OpenMP can have different orders in the front[0]
177  // so let's sort it so that the algorithm is deterministic
178  // given a seed
179  std::sort(front[0].begin(), front[0].end());
180 
181  int fi = 1;
182  while (front[fi-1].size() > 0) {
183 
184  std::vector<int>& fronti = front[fi-1];
185  std::vector<int> Q;
186  for (int i = 0; i < fronti.size(); ++i) {
187 
188  Individual& p = pop.individuals[fronti[i]];
189 
190  for (int j = 0; j < p.dominated.size() ; ++j) {
191 
192  Individual& q = pop.individuals[p.dominated[j]];
193  q.dcounter -= 1;
194 
195  if (q.dcounter == 0) {
196  q.set_rank(fi+1);
197  Q.push_back(p.dominated[j]);
198  }
199  }
200  }
201 
202  fi += 1;
203  front.push_back(Q);
204  }

208  void Pareto::crowding_distance(Population& pop, int fronti)
209  {
210  std::vector<int> F = front[fronti];
211  if (F.size() == 0 ) return;
212 
213  const int fsize = F.size();
214 
215  for (int i = 0; i < fsize; ++i)
216  pop.individuals[F[i]].crowd_dist = 0;
217 
218 
219  const int limit = pop[0].obj.size();
220  for (int m = 0; m < limit; ++m) {
221 
222  std::sort(F.begin(), F.end(), comparator_obj(pop,m));
223 
224  // in the paper dist=INF for the first and last, in the code
225  // this is only done to the first one or to the two first when size=2
226  pop.individuals[F[0]].crowd_dist = std::numeric_limits<double>::max();
227  if (fsize > 1)
228  pop.individuals[F[fsize-1]].crowd_dist = std::numeric_limits<double>::max();
229 
230  for (int i = 1; i < fsize-1; ++i)
231  {
232  if (pop.individuals[F[i]].crowd_dist != std::numeric_limits<double>::max())
233  { // crowd over obj
234  pop.individuals[F[i]].crowd_dist +=
235  (pop.individuals[F[i+1]].obj[m] - pop.individuals[F[i-1]].obj[m])
236  / (pop.individuals[F[fsize-1]].obj[m] - pop.individuals[F[0]].obj[m]);
237  }
238  }
239  }
240  }
