183  void Fewtwo::fit(MatrixXd& X, VectorXd& y)
184  {
204  // start the clock
205  timer.Reset();
206 
207  // define terminals based on size of X
208  params.set_terminals(X.rows());
209 
210  // initial model on raw input
211  params.msg("Fitting initial model", 1);
212  initial_model(X,y);
213  params.msg("Initial score: " + std::to_string(best_score), 1);
214 
215  // initialize population
216  params.msg("Initializing population", 1);
217  p_pop->init(best_ind,params);
218  params.msg("Initial population:\n"+p_pop->print_eqns(","),2);
219 
220  // resize F to be twice the pop-size x number of samples
221  F.resize(X.cols(),int(2*params.pop_size));
222 
223  // evaluate initial population
224  params.msg("Evaluating initial population",1);
225  p_eval->fitness(*p_pop,X,y,F,params);
226 
227  vector<size_t> survivors;
228 
229  // main generational loop
230  for (unsigned int g = 0; g<params.gens; ++g)
231  {
232 
233  // select parents
234  params.msg("selection..", 2);
235  vector<size_t> parents = p_sel->select(*p_pop, F, params);
236  params.msg("parents:\n"+p_pop->print_eqns(","), 2);
237 
238  // variation to produce offspring
239  params.msg("variation...", 2);
240  p_variation->vary(*p_pop, parents, params);
241  params.msg("offspring:\n" + p_pop->print_eqns(true), 2);
242 
243  // evaluate offspring
244  params.msg("evaluating offspring...", 2);
245  p_eval->fitness(*p_pop, X, y, F, params, true);
246 
247  // select survivors from combined pool of parents and offspring
248  params.msg("survival", 2);
249  survivors = p_surv->survive(*p_pop, F, params);
250 
251  // reduce population to survivors
252  params.msg("shrinking pop to survivors...",2);
253  p_pop->update(survivors);
254  params.msg("survivors:\n" + p_pop->print_eqns(), 2);
255 
256  update_best();
257  if (params.verbosity>0) print_stats(g+1);
258  }
259  params.msg("finished",1);
260  params.msg("best representation: " + best_ind.get_eqn(),1);
261  params.msg("score: " + std::to_string(best_score), 1);
262  }

264  void Fewtwo::initial_model(MatrixXd& X, VectorXd& y)
265  {
269  bool pass = true;
270  VectorXd yhat = p_eval->out_ml(X,y,params,pass,p_ml);
271 
272  // set terminal weights based on model
273  params.set_term_weights(p_ml->get_weights());
274 
275  if (params.classification) // assign best score as mean accuracy
276  best_score = (yhat.cast<int>().array() != y.cast<int>().array()).cast<double>().mean();
277  else // assign best score as MSE
278  best_score = (yhat-y).array().pow(2).mean();
279 
280  // initialize best_ind to be all the features
281  best_ind = Individual();
282  for (unsigned i =0; i<X.rows(); ++i)
283  best_ind.program.push_back(params.terminals[i]);
284  best_ind.fitness = best_score;
285  }

324  void Fewtwo::print_stats(unsigned int g)
325  {
326  unsigned num_models = std::min(20,p_pop->size());
327  double med_score = median(F.colwise().mean().array()); // median loss
328  ArrayXd Sizes(p_pop->size()); unsigned i = 0; // collect program sizes
329  for (const auto& p : p_pop->individuals){ Sizes(i) = p.size(); ++i;}
330  double med_size = median(Sizes); // median program size
331  string bar, space = ""; // progress bar
332  for (unsigned int i = 0; i<50; ++i){
333  if (i <= 50*g/params.gens) bar += "/";
334  else space += " ";
335  }
336  std::cout.precision(3);
337  std::cout << std::scientific;
338  std::cout << "Generation " << g << "/" << params.gens << " [" + bar + space + "]\n";
339  std::cout << "Min Loss\tMedian Loss\tMedian Program Size\tTime (s)\n"
340  << best_score << "\t" << med_score << "\t" << med_size << "\t" << timer << "\n";
341  std::cout << "Representation Pareto Front--------------------------------------\n";
342  std::cout << "Rank\tComplexity\tLoss\tRepresentation\n";
343 
344  // printing 10 individuals from the pareto front
345  unsigned n = 1;
346  vector<size_t> f = p_pop->sorted_front(n);
347  vector<size_t> fnew(2,0);
348  while (f.size() < num_models && fnew.size()>1)
349  {
350  fnew = p_pop->sorted_front(++n);
351  f.insert(f.end(),fnew.begin(),fnew.end());
352  }
353 
354  for (unsigned j = 0; j < std::min(num_models,unsigned(f.size())); ++j)
355  {
356  std::cout << p_pop->individuals[f[j]].rank << "\t"
357  << p_pop->individuals[f[j]].complexity() << "\t" << (*p_pop)[f[j]].fitness
358  << "\t" << p_pop->individuals[f[j]].get_eqn() << "\n";
359  }
