258  void Fewtwo::fit(MatrixXd& X, VectorXd& y)
259  {
280  // define terminals based on size of X
281  params.set_terminals(X.rows());
282 
283  // initial model on raw input
284  params.msg("Fitting initial model", 1);
285  initial_model(X,y);
286 
287 
288  // initialize population
289  params.msg("Initializing population", 1);
290  p_pop->init(params);
291 
292  // resize F to be twice the pop-size x number of samples
293  F.resize(X.cols(),int(2*params.pop_size));
294 
295  // evaluate initial population
296  params.msg("Evaluating initial population",1);
297  p_eval->fitness(*p_pop,X,y,F,params);
298 
299  vector<size_t> survivors;
300 
301  // main generational loop
302  for (size_t g = 0; g<params.gens; ++g)
303  {
304  params.msg("g " + std::to_string(g),1);
305 
306  // select parents
307  params.msg("selection", 1);
308  vector<size_t> parents = p_sel->select(F, params, r);
309 
310  // variation to produce offspring
311  params.msg("variation", 1);
312  p_variation->vary(*p_pop, parents, params);
313 
314  // evaluate offspring
315  params.msg("evaluation", 1);
316  p_eval->fitness(*p_pop, X, y, F, params);
317 
318  // select survivors from combined pool of parents and offspring
319  params.msg("survival", 1);
320  survivors = p_surv->select(F, params, r);
321 
322  // reduce population to survivors
323  p_pop->update(survivors);
324  }
325  }
