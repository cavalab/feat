257  void Fewtwo::fit(MatrixXd& X, VectorXd& y)
258  {
279  // define terminals based on size of X
280  params.set_terminals(X.rows());
281 
282  // initial model on raw input
283  params.msg("Fitting initial model", 1);
284  initial_model(X,y);
285 
286 
287  // initialize population
288  params.msg("Initializing population", 1);
289  p_pop->init(params);
290 
291  // resize F to be twice the pop-size x number of samples
292  F.resize(X.cols(),int(2*params.pop_size));
293 
294  // evaluate initial population
295  params.msg("Evaluating initial population",1);
296  p_eval->fitness(*p_pop,X,y,F,params);
297 
298  vector<size_t> survivors;
299 
300  // main generational loop
301  for (size_t g = 0; g<params.gens; ++g)
302  {
303  params.msg("g " + std::to_string(g),1);
304 
305  // select parents
306  params.msg("selection", 1);
307  vector<size_t> parents = p_sel->select(F, params, r);
308 
309  // variation to produce offspring
310  params.msg("variation", 1);
311  p_variation->vary(*p_pop, parents, params);
312 
313  // evaluate offspring
314  params.msg("evaluation", 1);
315  p_eval->fitness(*p_pop, X, y, F, params);
316 
317  // select survivors from combined pool of parents and offspring
318  params.msg("survival", 1);
319  survivors = p_surv->select(F, params, r);
320 
321  // reduce population to survivors
322  p_pop->update(survivors);
323  }
324  }
